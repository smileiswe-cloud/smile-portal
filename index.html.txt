    <script>
        let timeout = null;
        
        const telegram = window.Telegram.WebApp;
        telegram.ready();
        telegram.expand();
        
        const user = telegram.initDataUnsafe?.user;
        const chatId = user?.id;
        
        // ⚠️ IMPORTANT: Replace with your Apps Script URL
        const API_URL = 'https://script.google.com/macros/s/AKfycbxj9Z1I7sd9zFuVOjIVRfpo7aJKWbZY1hBNVqo0KYK9F0bgI2RFCr1rAdU5tDk4ATU/exec';
        
        const searchInput = document.getElementById('searchInput');
        const resultsDiv = document.getElementById('results');
        const resultsCountDiv = document.getElementById('resultsCount');
        const backToTopBtn = document.getElementById('backToTop');
        
        searchInput.addEventListener('input', function(e) {
            clearTimeout(timeout);
            timeout = setTimeout(() => {
                searchProducts(e.target.value);
            }, 400);
        });
        
        document.querySelectorAll('.cat-btn').forEach(btn => {
            btn.addEventListener('click', function() {
                const category = this.getAttribute('data-cat');
                searchInput.value = category;
                searchProducts(category);
            });
        });
        
        backToTopBtn.addEventListener('click', () => {
            window.scrollTo({ top: 0, behavior: 'smooth' });
        });
        
        window.addEventListener('scroll', () => {
            if (window.scrollY > 300) {
                backToTopBtn.classList.remove('hidden');
            } else {
                backToTopBtn.classList.add('hidden');
            }
        });
        
        async function searchProducts(query) {
            if (!query.trim()) {
                showEmptyState();
                return;
            }
            
            showLoading();
            
            try {
                const response = await fetch(API_URL, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ 
                        action: 'search', 
                        query: query,
                        chatId: chatId 
                    })
                });
                const data = await response.json();
                displayResults(data, query);
            } catch(err) {
                console.error(err);
                showError();
            }
        }
        
        function displayResults(products, query) {
            if (!products || products.length === 0) {
                resultsDiv.innerHTML = `<div class="empty-state">❌ "${escapeHtml(query)}" နှင့် ဆက်စပ်သော ပစ္စည်း မတွေ့ပါ။</div>`;
                resultsCountDiv.style.display = 'none';
                return;
            }
            
            resultsCountDiv.style.display = 'block';
            resultsCountDiv.innerHTML = `📊 တွေ့ရှိစုစုပေါင်း: ${products.length} ခု`;
            
            let html = '';
            for (let p of products) {
                const stockClass = p.stock <= 5 ? 'low' : '';
                html += `
                    <div class="product-card" onclick="viewProduct('${escapeHtml(p.skid)}')">
                        <div class="product-name">📱 ${escapeHtml(p.name)}</div>
                        <div class="product-price">💰 ${formatPrice(p.price)} Ks</div>
                        <div class="product-stock ${stockClass}">📦 လက်ကျန်: ${p.stock} ခု</div>
                        <div class="product-skid">🔑 SKID: ${escapeHtml(p.skid)}</div>
                    </div>
                `;
            }
            resultsDiv.innerHTML = html;
        }
        
        function viewProduct(skid) {
            telegram.sendData(JSON.stringify({ action: 'view', skid: skid }));
        }
        
        function showLoading() {
            resultsDiv.innerHTML = '<div class="loading-state">⏳ ရှာဖွေနေပါသည်...</div>';
            resultsCountDiv.style.display = 'none';
        }
        
        function showEmptyState() {
            resultsDiv.innerHTML = '<div class="empty-state">📱 ပစ္စည်းအမည် (သို့) SKID ရိုက်ထည့်ပါ။</div>';
            resultsCountDiv.style.display = 'none';
        }
        
        function showError() {
            resultsDiv.innerHTML = '<div class="empty-state">❌ ချိတ်ဆက်မှု အမှားရှိပါသည်။ နောက်မှ ထပ်ကြိုးစားပါ။</div>';
            resultsCountDiv.style.display = 'none';
        }
        
        function formatPrice(price) {
            return price.toString().replace(/\B(?=(\d{3})+(?!\d))/g, ",");
        }
        
        function escapeHtml(str) {
            if (!str) return '';
            return str.replace(/[&<>]/g, function(m) {
                if (m === '&') return '&amp;';
                if (m === '<') return '&lt;';
                if (m === '>') return '&gt;';
                return m;
            });
        }
